### Return Type Overloading Resolution
```C++
struct A {
	operator float() {return 3.0f;}
	operator int() {return 2;}
}

auto f() {
	return A{ };
}

float x = f();
int y = f();
cout << x << "" << y;
```
### Increment and Decrement Operator
```C++
struct A{
int* ptr;
int pos;
A& operator++(){
++ptr;
++pos;
return *this;
}

A operator++(int a){
	A tmp = *this
	++ptr;
	++pos;
	return tmp;
	}
};
```
